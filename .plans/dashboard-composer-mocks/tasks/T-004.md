```typescript
import { http, HttpResponse } from "msw";
import type { ListSessionsOutput, ListModelsOutput, GetMessagesOutput, SessionStateOutput } from "../../src/api-types.js";

export const handlers = [
  // GET /api/sessions.list → sesiones mock
  http.post("*/api/sessions.list", () => {
    return HttpResponse.json<ListSessionsOutput>({ sessions: [...] });
  }),

  // POST /api/sessions.prompt → success
  http.post("*/api/sessions.prompt", () => {
    return HttpResponse.json({ success: true });
  }),

  // POST /api/sessions.models.list → modelos mock
  http.post("*/api/sessions.models.list", () => {
    return HttpResponse.json<ListModelsOutput>({ models: [...], current: {...} });
  }),

  // POST /api/sessions.models.set → success

    return HttpResponse.json({ success: true });
  }),

  // POST /api/sessions.messages.get → mensajes mock
  http.post("*/api/sessions.messages.get", () => {
    return HttpResponse.json<GetMessagesOutput>({ messages: [...] });

  // POST /api/sessions.create → nueva sesión
  http.post("*/api/sessions.create", () => {
```

```typescript
/**
 * Creates an async generator that yields a sequence of SSE events.
 * Useful for testing streaming message handling.
 */
export async function* createMockSSEStream(events: ServerEvent[]): AsyncIterableIterator<ServerEvent> {
  for (const event of events) {
    await new Promise((r) => setTimeout(r, 5)); // Simulate network delay
    yield event;
  }
```

### `test/test-utils.tsx`
Wrapper de testing que provee todos los providers:

```typescript
import React, { type ReactNode } from "react";
import { render, type RenderOptions } from "@testing-library/react";
import { SessionRuntimeProvider } from "../src/store/provider.js";
// import { UIProvider } from "../src/store/ui/provider.js"; // Una vez creado

interface CustomRenderOptions extends Omit<RenderOptions, "wrapper"> {
  // Opciones adicionales: sessionId inicial, UI state inicial, etc.
  initialSessionId?: string | null;
}

function AllTheProviders({ children, initialSessionId }: { children: ReactNode; initialSessionId?: string | null }) {
  return (
    <SessionRuntimeProvider>
      {/* <UIProvider initialSessionId={initialSessionId}> */}
        {children}
      {/* </UIProvider> */}
    </SessionRuntimeProvider>
  );
}

export function renderWithProviders(
  ui: React.ReactElement,
  options?: CustomRenderOptions,
) {
  const { initialSessionId, ...renderOptions } = options ?? {};
  return render(ui, {
    wrapper: ({ children }) => (
      <AllTheProviders initialSessionId={initialSessionId}>
        {children}
      </AllTheProviders>
    ),
    ...renderOptions,
  });
}

// Re-exportar todo desde testing-library para conveniencia
export * from "@testing-library/react";
export { default as userEvent } from "@testing-library/user-event";
```

## Notas
- MSW intercepta requests HTTP a nivel de Node, ideal para mockear oRPC
- El mock de oRPC (`createMockORPCClient`) permite tests sin MSW — útil para tests unitarios
- Las fábricas de datos usan funciones puras — fáciles de personalizar por test
- `renderWithProviders` debe actualizarse cuando T-001 (UIProvider) esté completo
- Para SSE, como oRPC usa `AsyncIterableIterator`, el mock de SSE se puede inyectar directamente en el runtime sin pasar por MSW
